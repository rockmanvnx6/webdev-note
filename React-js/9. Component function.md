# 9. Component function

Can put this in side `createClass`

for example:

```react
export const SkiDayCount = createClass({
    percentToDecimal(decimal) {
        return (decimal*100) + '%';
    },

    calcGoalProgress(total,goal) {
        return this.percentToDecimal(total/goal);
    },

    render() {
        return (
            <div className="ski-day-count">
                <div className="total-days">
                    <span>{this.props.total}</span>
                    <span>days</span>
                </div>
                <div className="power-days">
                    <span>{this.props.powder}</span>
                    <span>days</span>
                </div>
                <div className="backcountry-days">
                    <span>{this.props.backcountry}</span>
                    <span>hiking day</span>
                </div>
                <div>
                    <span>{this.calcGoalProgress(
                        this.props.total, 
                        this.props.goal
                        )}</span>
                </div>
            </div>
        )
    }
});
```

use `this.` to refer to a function in the same class.
